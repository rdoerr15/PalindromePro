function getValues() {
  let alertBox = document.getElementById("alert");
  alertBox.classList.remove("alert-danger");
  alertBox.classList.remove("alert-success");
  alertBox.classList.add("d-none");

  let userInput = document.getElementById("message").value;

  if (userInput.length > 0) {
    let reversedInput = reverseString(userInput);
    let isPalindrome = checkForPalindrome(userInput);

    displayResult(reversedInput, isPalindrome);
  } else {
    Swal.fire({
      icon: "error",
      backdrop: false,
      title: "PalindromePro",
      text: "You must enter an input to check for palindrome!",
    });
  }
}

function checkForPalindrome(userString) {
  userString = userString.toLowerCase();

  const regex = /[^a-z0-9]/gi;
  userString = userString.replace(regex, "");

  let reversedString = reverseString(userString);

  let isPalindrome = reversedString == userString;

  return isPalindrome;
}

function reverseString(message) {
  let reversedMsg = "";

  for (let i = message.length - 1; i >= 0; i--) {
    reversedMsg += message[i];
  }

  return reversedMsg;
}

function displayResult(reversedInput, isPalindrome) {
  let resultMessage = "";

  let alertClass = isPalindrome ? "alert-success" : "alert-danger";
  let alertBox = document.getElementById("alert")
  alertBox.classList.add(alertClass);

  if (isPalindrome) {
    resultMessage = "You entered a palindrome!";
  } else {
    resultMessage = "You did not enter a plaindrome!";
  }

  resultMessage += " Your message reversed is " + reversedInput;

  document.getElementById("msg").textContent = resultMessage;
  alertBox.classList.remove("d-none");

The code is structured into four separate functions. Below you can get a better understanding of how each function works.

getValues( )

This function is called when a user clicks the button to retrieve the input value. It starts by obtaining references to the necessary HTML elements, such as the alert box and the user input field. It removes any existing alert classes, adds the "d-none" class to hide the alert box, and retrieves the user's input value. If the input has a length greater than zero, it proceeds to call the reverseString() function to reverse the user's input and the checkForPalindrome() function to determine if it is a palindrome. Finally, it calls the displayResult() function, passing the reversed input and the result of the palindrome check.

checkForPalindrome(userString)

This function takes the userString parameter and checks if it is a palindrome. It first converts the string to lowercase using the toLowerCase() method to ensure case-insensitive comparison. It then defines a regular expression (regex) to match any characters that are not alphanumeric. It uses the replace() method with the regex to remove those non-alphanumeric characters from the userString. Next, it obtains the reversed string by calling the reverseString() function on the cleaned userString. Finally, it compares the reversed string with the original cleaned string and returns true if they are equal, indicating that the input is a palindrome, or false otherwise.

reverseString(message)

This function takes a message parameter and returns the reversed version of it. It initializes an empty string called reversedMsg. Then, it uses a for loop to iterate through each character of the message string, starting from the last character and moving backward. Within each iteration, it appends the current character to the reversedMsg string. After looping through all characters, it returns the reversedMsg string.

displayResult(reversedInput, isPalindrome)

This function is responsible for updating the result message and displaying it on the webpage. It takes two parameters: reversedInput (the reversed user input) and isPalindrome (a boolean indicating whether the input is a palindrome or not). It initializes the resultMessage variable as an empty string. Based on the value of isPalindrome, it sets resultMessage to either "You entered a palindrome!" or "You did not enter a palindrome!". It then appends the reversed input to the resultMessage. Next, it retrieves a reference to the HTML element representing the result message and the alert box. It adds the appropriate alert class (alert-success for palindromes, alert-danger for non-palindromes) to the alert box. It updates the text content of the result message element with the resultMessage string. Finally, it removes the "d-none" class from the alert box, making it visible to the user.
